\section{Installation}

\subsection{Requirements}

\begin{itemize}
    \item Python 3.10+
    \item NumPy 2.0+
    \item SciPy 1.10+
    \item Pandas 2.0+
    \item BioPython 1.80+
    \item Scikit-learn 1.3+ (for predictive model)
    \item Matplotlib 3.7+ (for visualization)
\end{itemize}

\subsection{Quick Start}

\begin{lstlisting}[language=bash]
git clone https://github.com/Adi-Baba/SHTDNA.git
cd SHTDNA
pip install -r requirements.txt
python scripts/extract_vij.py data/1bna.pdb
\end{lstlisting}

\section{Key Functions}

\subsection{\texorpdfstring{\code{compute\_sht(pdb\_file)}}{compute\_sht(pdb\_file)}}

Main entry point. Returns dictionary with Vij, Shambhian, Mamton for all basepairs.

\begin{lstlisting}[language=Python]
from sht_core import compute_sht

results = compute_sht('1bna.pdb')
print(results['vij_mean'])      # 0.073
print(results['shambhian_mean']) # 0.952
print(results['mamton_mean'])    # 168
\end{lstlisting}
